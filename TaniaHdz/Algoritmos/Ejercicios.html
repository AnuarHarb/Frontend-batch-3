      Inicio(dormir, despertar)
        leer dormir
        leer despertar

        Si dormir > 12 
          HorasDormidas = (24 - dormir) + despertar
        Sino
          HorasDormidas = despertar - dormir
        fin-si

        Escribe HorasDormidas
      Fin

Precio del super

      Inicio(sal, azúcar, cereal, arroz, helado) 
        leer sal
        leer azúcar
        leer cereal
        leer arroz
        leer helado

        PrecioSuper = sal + azúcar + cereal + arroz + helado

        Escribe PrecioSuper 

      Fin

Presupuesto

      Inicio(ingreso1, ingreso2, gasto1, gasto2, gasto3)
        leer ingreso1
        leer ingreso2
        leer gasto1
        leer gasto2
        leer gasto3

        Presupuesto = (ingreso1 + ingreso2) - (gasto1 + gasto2 + gasto3)

        Escribe Presupuesto

      Fin

Herencia de Nietos

      Inicio(jose, juan, herencia)
        leer jose
        leer juan
        leer herencia

        TotalDinero = jose + juan + herencia
        TotalJuan = juan + (herencia/2)
        TotalJose = jose + (herencia/2)

        Escribe TotalDinero
        Escribe TotalJuan
        Escribe TotalJose

      Fin

Restaurante

      Inicio(precio, individuos)
        leer precio
        leer individuos

        Propina = 1.10
        PagoIndividual = precio/individuos
        PagoPropina = PagoIndividual * Propina

        Escribe PagoIndividual
        Escribe PagoPropina
